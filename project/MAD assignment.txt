import 'package:flutter/material.dart';
void main() {
  runApp(QuestionAnalyzerApp());
}
class QuestionAnalyzerApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: AnalyzerScreen(),
    );
  }
}
class AnalyzerScreen extends StatefulWidget {
  @override
  State<AnalyzerScreen> createState() => _AnalyzerScreenState();
}
class _AnalyzerScreenState extends State<AnalyzerScreen> {
  final TextEditingController controller = TextEditingController()
  int totalQuestions = 0;
  int oneMark = 0;
  int twoMark = 0;
  int fiveMark = 0;
  int totalMarks = 0;
  void analyzePaper() {
    final text = controller.text.trim();
    if (text.isEmpty) return;
    final questions = text.split('\n');
    int tq = 0, o = 0, t = 0, f = 0, tm = 0;
    for (var q in questions) {
      final line = q.toLowerCase().trim();
      if (line.isNotEmpty) {
        tq++;
        if (line.contains("1 mark")) {
          o++;
          tm += 1;
        } else if (line.contains("2 mark")) {
          t++;
          tm += 2;
        } else if (line.contains("5 mark")) {
          f++;
          tm += 5;
        }
      }
    }
    setState(() {
      totalQuestions = tq;
      oneMark = o;
      twoMark = t;
      fiveMark = f;
      totalMarks = tm;
    });
  }
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("Question Paper Analyzer"),
        actions: [
          Padding(
            padding: const EdgeInsets.only(right: 12),
            child: Column(
              mainAxisAlignment: MainAxisAlignment.center,
              crossAxisAlignment: CrossAxisAlignment.end,
              children: const [
                Text(
                  "Gubba Sri Vidyambika",
                  style: TextStyle(fontSize: 12, fontWeight: FontWeight.bold),
                ),
                Text(
                  "Roll No: 160124737147",
                  style: TextStyle(fontSize: 12),
                ),
              ],
            ),
          )
        ],
      ),
      body: SingleChildScrollView(
        child: Padding(
          padding: const EdgeInsets.all(16),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              TextField(
                controller: controller,
                maxLines: 8,
                decoration: InputDecoration(
                  border: OutlineInputBorder(),
                  hintText: "Paste question paper here...",
                ),
              ),
           const SizedBox(height: 12),
              Center(
                child: ElevatedButton(
                  onPressed: analyzePaper,
                  child: const Text("Analyze"),
                ),
              ),
              const SizedBox(height: 20),
              const Text(
                "Results:",
                style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
              ),
              const SizedBox(height: 10),
              Text("Total Questions: $totalQuestions"),
              Text("Total Marks: $totalMarks"),
              Text("1 Mark Questions: $oneMark"),
              Text("2 Mark Questions: $twoMark"),
              Text("5 Mark Questions: $fiveMark"),
            ],
          ),
        ),
      ),
    );
 }
}
